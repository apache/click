Control Interface

 In Click Controls provide the server side components that process user input, and render their display to the user. Controls are equivalent to Visual Basic Controls or Delphi Components.

Controls handle the processing of user input with the onProcess method and rendering their HTML display using the toString() method. The sequence diagram below illustrates a Control being processed and rendering its display.

public class SimpleCallback extends Page {

    public void onInit() {
        ActionLink clickLink = new ActionLink("onClick");
        clickLink.setListener(this, "onClick");
        addControl(goLink);    
    }
    
    public boolean onClick() {
        System.out.println("onClick invoked");
        return true;
    }
}

public class LinkPage extends Page {

    public void onInit() {
        ActionLink goLink = new ActionLink("goLink");
        goLink.setListener(this, "onGoClick");
        addControl(goLink);
        
        ActionLink continueLink = new ActionLink("stopLink");
        stopLink.setListener(this, "onStopClick");
        addControl(stopLink);       
    }
    
    public boolean onGoClick() {
        System.out.println("onGoClick invoked");
        return true;
    }
    
    public boolean onStopClick() {
        System.out.println("onLinkClick invoked");
        return false;
    }
    
    public void onGet() {

    }
}
